        import { initializeApp } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-app.js";
        import { getAuth, signInWithPopup, GoogleAuthProvider, signOut, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-auth.js";
        import { getFirestore, doc, setDoc, getDoc } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-firestore.js";

        // ==========================================
        // 【重要】あなたのFirebase設定に書き換えてください
        // ==========================================
        const firebaseConfig = {
            apiKey: "YOUR_API_KEY",
            authDomain: "YOUR_PROJECT_ID.firebaseapp.com",
            projectId: "YOUR_PROJECT_ID",
            storageBucket: "YOUR_PROJECT_ID.appspot.com",
            messagingSenderId: "YOUR_SENDER_ID",
            appId: "YOUR_APP_ID"
        };

        // Firebase初期化
        const app = initializeApp(firebaseConfig);
        const auth = getAuth(app);
        const db = getFirestore(app);
        const provider = new GoogleAuthProvider();

        let currentUser = null;

        // グローバルにFirebase関数を露出（HTMLのonclick属性から呼ぶため）
        window.loginWithGoogle = async function() {
            try {
                await signInWithPopup(auth, provider);
            } catch (error) {
                console.error("Googleログインエラー:", error);
                alert("ログインに失敗しました。詳細: " + error.message);
            }
        };

        window.logout = async function() {
            if (confirm("ログアウトしますか？")) {
                try {
                    await signOut(auth);
                    location.reload();
                } catch (error) {
                    console.error("ログアウトエラー:", error);
                }
            }
        };

        // バックグラウンド同期保存用のラッパー
        window.saveDataToFirestore = async function() {
            if (!currentUser) return;
            try {
                const docRef = doc(db, "users", currentUser.uid);
                // アプリのグローバルデータ appData をそのまま丸ごと同期保存
                await setDoc(docRef, window.appData);
                console.log("Firestoreに同期保存しました。");
            } catch (error) {
                console.error("Firestore自動保存エラー:", error);
            }
        };

        // ログイン状態の監視
        onAuthStateChanged(auth, async (user) => {
            const authContainer = document.getElementById('firebase-auth-container');
            const badge = document.getElementById('save-status-badge');
            
            if (user) {
                currentUser = user;
                // バッジの表示切り替え
                if (badge) {
                    badge.className = "bg-indigo-600 text-[10px] text-white px-3 py-1 rounded-full font-bold flex items-center gap-1.5 shadow-sm";
                    badge.innerHTML = `<span class="w-1.5 h-1.5 rounded-full bg-white animate-pulse"></span>自動保存: ON (クラウド保存)`;
                }
                // アカウントUI挿入
                authContainer.innerHTML = `
                    <div class="flex items-center gap-2 bg-white/10 p-1 rounded-lg backdrop-blur-xs border border-white/10">
                        <img src="${user.photoURL}" class="w-5 h-5 rounded-full border border-white/20" alt="avatar" referrerpolicy="no-referrer">
                        <span class="text-[10px] font-bold hidden sm:inline">${user.displayName}</span>
                        <button onclick="logout()" class="text-[9px] bg-white/20 hover:bg-rose-500 hover:text-white text-white px-2 py-0.5 rounded transition font-bold">ログアウト</button>
                    </div>
                `;

                // クラウドからデータ読み込み
                try {
                    const docRef = doc(db, "users", user.uid);
                    const docSnap = await getDoc(docRef);
                    if (docSnap.exists()) {
                        window.appData = docSnap.data();
                        console.log("Firestoreからデータをロードしました。");
                    } else {
                        console.log("ユーザーデータがクラウドにありません。初期化します。");
                        window.initAppData();
                        await window.saveDataToFirestore();
                    }
                    window.renderApp();
                } catch (e) {
                    console.error("データロードエラー:", e);
                    window.loadFromLocalStorage();
                }
            } else {
                currentUser = null;
                if (badge) {
                    badge.className = "bg-emerald-500 text-[10px] text-white px-3 py-1 rounded-full font-bold flex items-center gap-1.5 shadow-sm";
                    badge.innerHTML = `<span class="w-1.5 h-1.5 rounded-full bg-white animate-pulse"></span>自動保存: ON (ブラウザ保存)`;
                }
                authContainer.innerHTML = `
                    <button onclick="loginWithGoogle()" class="bg-white hover:bg-slate-100 text-slate-800 text-[10px] font-bold px-3 py-1 rounded-full transition flex items-center gap-1 shadow-sm">
                        <svg class="w-3 h-3" viewBox="0 0 24 24"><path fill="currentColor" d="M21.35 11.1H12v2.7h5.38c-.23 1.28-1.03 2.37-2.11 3.1v2.56h3.42c2 1.84 3.16 4.54 3.16 7.69c0-.53-.05-1.05-.15-1.56z" color="#4285f4"/><path fill="currentColor" d="M12 23c2.97 0 5.46-.98 7.28-2.66l-3.42-2.56c-.95.63-2.16 1.02-3.86 1.02c-2.97 0-5.48-2-6.38-4.7H2.1v2.66C3.93 20.19 7.73 23 12 23z" color="#34a853"/><path fill="currentColor" d="M5.62 14.1a6.76 6.76 0 0 1 0-4.2V7.24H2.1a11.12 11.12 0 0 0 0 9.52l3.52-2.66z" color="#fbbc05"/><path fill="currentColor" d="M12 4.75c1.62 0 3.08.56 4.22 1.65l3.15-3.15C17.45 1.44 14.96 0 12 0C7.73 0 3.93 2.81 2.1 6.81l3.52 2.66c.9-2.7 3.41-4.72 6.38-4.72z" color="#ea4335"/></svg>

        ];

        // --- 初期化 ---
        function initAppData() {
            const now = new Date();
            appData.currentYear = now.getFullYear();
            appData.currentMonth = now.getMonth() + 1;
            appData.subjects = [
                { id: 'subj-1', name: '国語', color: '#ef4444' },
                { id: 'subj-2', name: '数学', color: '#3b82f6' }
            ];
            appData.dailyTimelines = {};
            appData.monthlyLogs = {};
            appData.myTimers = {
                focus: [15, 25, 45, 60, 90],
                break: [3, 5, 10, 15, 30]
            };
            window.appData = appData;
        }
        window.initAppData = initAppData;

        // 保存のトリガー（LocalStorageとFirestoreの共通化）
        function saveAllData() {
            localStorage.setItem('study_scheduler_data_v4', JSON.stringify(appData));
            // もしFirebase用の同期関数があれば裏側で送信
            if (typeof window.saveDataToFirestore === 'function') {
                window.saveDataToFirestore();
            }
        }

        function loadFromLocalStorage() {
            const saved = localStorage.getItem('study_scheduler_data_v4');
            if (saved) {
                try {
                    appData = JSON.parse(saved);
                    window.appData = appData;
                    // 年月のバリデーション
                    if(!appData.currentYear || !appData.currentMonth) {
                        const d = new Date();
                        appData.currentYear = d.getFullYear();
                        appData.currentMonth = d.getMonth() + 1;
                    }
                    if(!appData.myTimers) {
                        appData.myTimers = { focus: [15,25,45,60,90], break: [3,5,10,15,30] };
                    }
                } catch (e) {
                    initAppData();
                }
            } else {
                initAppData();
            }
            renderApp();
        }
        window.loadFromLocalStorage = loadFromLocalStorage;
